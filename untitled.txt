--// LOOP BRING ALL Menu (with ON/OFF + Reset Position)
-- LocalScript

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local LocalPlayer = Players.LocalPlayer

-- Vars
local bringAllEnabled = false
local bringPos = LocalPlayer.Character and (LocalPlayer.Character:WaitForChild("HumanoidRootPart").Position - Vector3.new(5,0,0))

-- GUI
local gui = Instance.new("ScreenGui")
gui.Name = "LoopBringAllGui"
gui.Parent = LocalPlayer:WaitForChild("PlayerGui")
gui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
gui.Enabled = false

local frame = Instance.new("Frame", gui)
frame.Size = UDim2.new(0,300,0,160)
frame.Position = UDim2.new(0.5,-150,0.4,-80)
frame.BackgroundColor3 = Color3.fromRGB(20,20,20)
frame.Active = true
frame.Draggable = true
Instance.new("UICorner", frame).CornerRadius = UDim.new(0,12)

local title = Instance.new("TextLabel", frame)
title.Size = UDim2.new(1,0,0,40)
title.BackgroundTransparency = 1
title.Text = "88K$HITZ LOOPBRING ALL"
title.Font = Enum.Font.GothamBold
title.TextSize = 20
title.TextColor3 = Color3.fromRGB(255,255,255)

-- Toggle Button
local toggleBtn = Instance.new("TextButton", frame)
toggleBtn.Size = UDim2.new(0.9,0,0,40)
toggleBtn.Position = UDim2.new(0.05,0,0,60)
toggleBtn.BackgroundColor3 = Color3.fromRGB(60,0,0)
toggleBtn.Text = "OFF"
toggleBtn.Font = Enum.Font.GothamBold
toggleBtn.TextColor3 = Color3.fromRGB(255,255,255)
toggleBtn.TextScaled = true
Instance.new("UICorner",toggleBtn).CornerRadius = UDim.new(0,8)

-- Reset Button
local resetBtn = Instance.new("TextButton", frame)
resetBtn.Size = UDim2.new(0.9,0,0,40)
resetBtn.Position = UDim2.new(0.05,0,0,110)
resetBtn.BackgroundColor3 = Color3.fromRGB(40,40,40)
resetBtn.Text = "RESET POSITION"
resetBtn.Font = Enum.Font.GothamBold
resetBtn.TextColor3 = Color3.fromRGB(255,255,255)
resetBtn.TextScaled = true
Instance.new("UICorner",resetBtn).CornerRadius = UDim.new(0,8)

-- Functions
local function setToggle(state)
	bringAllEnabled = state
	if bringAllEnabled then
		toggleBtn.Text = "ON"
		toggleBtn.BackgroundColor3 = Color3.fromRGB(0,150,0)
	else
		toggleBtn.Text = "OFF"
		toggleBtn.BackgroundColor3 = Color3.fromRGB(60,0,0)
	end
end

toggleBtn.MouseButton1Click:Connect(function()
	setToggle(not bringAllEnabled)
end)

resetBtn.MouseButton1Click:Connect(function()
	local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
	if hrp then
		bringPos = hrp.Position - Vector3.new(5,0,0)
	end
end)

-- Bring Loop
RunService.RenderStepped:Connect(function()
	if bringAllEnabled and bringPos then
		for _,plr in ipairs(Players:GetPlayers()) do
			if plr ~= LocalPlayer and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
				-- Remove tools
				plr.Backpack:ClearAllChildren()
				for _,tool in ipairs(plr.Character:GetChildren()) do
					if tool:IsA("Tool") then
						tool:Destroy()
					end
				end
				-- Teleport near localplayer
				plr.Character.HumanoidRootPart.CFrame = CFrame.new(bringPos)
			end
		end
	end
end)

-- Function para tawagin mula Rayfield button
local function ShowLoopBringAll()
	gui.Enabled = true
end

MAINTab:CreateButton({
    Name = "LOOPBRING ALL [88K]",
    Callback = function()
        ShowLoopBringAll()
   end,
})
