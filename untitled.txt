-- Syntax Highlight Function
local keywords = {"local","function","end","if","then","else","elseif","for","while","do","return","game","workspace","Players"}
local function updateHighlight()
    local src = editorBox.Text
    local colored = src
    for _, kw in ipairs(keywords) do
        colored = colored:gsub("%f[%w]"..kw.."%f[%W]", '<font color="rgb(85,170,255)">'..kw.."</font>")
    end
    colored = colored:gsub("\"(.-)\"", '<font color="rgb(255,170,0)">"%1"</font>')
    colored = colored:gsub("%-%-.-\n", '<font color="rgb(0,200,0)">%1</font>\n')
    colored = colored:gsub("(%d+)", '<font color="rgb(180,220,180)">%1</font>')
    highlight.Text = colored
    if tabs[currentTab] then
        tabs[currentTab].text = editorBox.Text
        tabs[currentTab].highlight = colored
    end
end
